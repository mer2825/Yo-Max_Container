Fecha Tipo de Egreso Comentario Registrado Por Monto Acciones
@GetMapping("/api/listar") @ResponseBody public ResponseEntity listarEgresosApi( @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate desde, @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate hasta) { try { LocalDate fechaDesde = desde != null ? desde : LocalDate.now().withDayOfMonth(1); LocalDate fechaHasta = hasta != null ? hasta : LocalDate.now(); List egresos = egresoService.listarEgresosDesdeMovimientosCaja(fechaDesde, fechaHasta); Map response = new HashMap<>(); response.put("success", true); response.put("data", egresos); return ResponseEntity.ok(response); } catch (Exception e) { logger.error("Error al listar egresos", e); Map response = new HashMap<>(); response.put("success", false); response.put("message", "Error al listar egresos: " + e.getMessage()); return ResponseEntity.badRequest().body(response); } }
